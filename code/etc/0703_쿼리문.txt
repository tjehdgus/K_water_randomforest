drop table dw;

create table dw
as
select d.지점명,d.일시,w.일강수량,d.spi6
    from drought d,water w
    where d.일시 = w.일시 and d.지점명 = w.지점명;
drop table a;

create table a
as
select rownum as id,일강수량,최고기온,평균상대습도,평균지면온도,d.spi6,to_number(합계소형증발량) as 합계소형증발량,방류량,사용가능량
    from dw d,drought_2024 d2 ,dam d3
    where d.일시 = d2.일시 and d.지점명 = d2.지점명
    and d3.날짜 = d2.일시 and d3.지점 = d2.지점명;

select corr(합계소형증발량,spi6)
    from a;
    
delete from a where 합계소형증발량 is null;
